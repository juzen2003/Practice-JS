### Object.keys/values/entries
```js
const obj = {"a": 2, "b": 4, "c": 6}
Object.keys(obj)
// ["a", "b", "c"]
Object.values(obj)
// [2, 4, 6]
Object.entries(obj)
// [ [ 'a', 1 ], [ 'b', 2 ], [ 'c', 3 ] ]
obj["a"]
// 2
obj["x"]
// undefined
```
### Check if POJO is empty
```js
Object.keys(obj).length === 0 // empty
Object.keys(obj).length !== 0 // not empty
```
### for...in
```js
let hsh = {'a': 1, 'b': 2};
for(let a in hsh) {
  console.log(a);
  console.log(hsh[a]);
}
// a
// 1
// b
// 2
```

### concat & push
```js
let arr = [1,2,3]
arr.push(4)
// arr = [1,2,3,4]
arr.push([5])
// arr = [1,2,3,4,[5]]
arr.concat(3)
// arr = [1,2,3,4,[5]], arr not change, need to re-assign
arr = arr.concat(3)
// arr = [1,2,3,4,[5],3]
```

### array sort
```js
let arr = ["aaple","orange","banana"];
arr.sort();
// [ 'apple', 'banana', 'orange' ]
let arr2 = [1,5,3];
arr2.sort();
// [1,3,5]
arr2.sort(function(a,b){return b-a});
// b > a return 1, b < a return -1, b = a return 0
// [5,3,1]
```
### string slice
start index is included, end index is not included
```js
"cbbd".slice(1,2);
// 'b'
"cbbd".slice(1,3);
// 'bb'
```
### includes

```js
// true

// false
[1,2,3].includes(1);
// true

```
### array initialization
```js
let arr = new Array(3);
// let arr = Array(3); would work as well
// initialize arr with length 3
// [ <3 empty items> ]
arr.fill("", 0);
// fill all elements with "", starting from idx 0
// ["", "", ""]
arr.fill("x",0,2);
// value, startIdx, endIdx
// fill idx 0 up to 2 (not including 2) with "x"
// ["x", "x", ""]
```
